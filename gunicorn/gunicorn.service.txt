[Unit]
Description=Gunicorn instance for project app
After=network.target

[Service]
User=www-data
Group=www-data
WorkingDirectory=/opt/project
Environment="PATH=/opt/project/.venv/bin"
ExecStart=/opt/project/.venv/bin/gunicorn -w 4 -b 127.0.0.1:8000 app:app
Restart=always

[Install]
WantedBy=multi-user.target
